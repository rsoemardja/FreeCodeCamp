<style type='text/sass'>
// In sass it is much more case and context sensitive in how you sort your characters
// Here we are are learning to apply a style with a @while loop in sass
$x: 1;
@while $x < 11 {
  .text-#{$x} {
  font-size: 15px * $x;
  }
  $x: $x + 1;
}

</style>

<p class="text-1">Hello</p>
<p class="text-2">Hello</p>
<p class="text-3">Hello</p>
<p class="text-4">Hello</p>
<p class="text-5">Hello</p>
